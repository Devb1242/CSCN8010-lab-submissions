def prediction_model(model, x_image):
    y_pred = np.argmax(model.predict(x_image.reshape((-1, 28 * 28))), axis=1)
    return y_pred

def model_evaluation(y_pred , y_test):
    report = classification_report(y_test, y_pred,output_dict=True)
    return (report)
